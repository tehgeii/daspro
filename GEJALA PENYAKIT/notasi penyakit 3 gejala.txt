//Judul: deteksi dan diagnosa penyakit berdasarkan gejala
// Kamus inputan dan karakter
pilihan <-- char
// Kamus untuk gejala
keluhan1, keluhan2, keluhan3, keluhan4 <-- char
// Diskripsi
repeat
{
	output "DETEKSI & DIAGNOSA"
        output "Silakan ceritakan bagaimana perasaan Anda!"
        output "A = SESAK NAPAS"
        output "B = RADANG"
        output "C = PUSING"
        output "D = PANAS DINGIN"
        output "E = SAKIT PERUT"

        output "Masukkan 3 gejala yang Anda rasakan (A/B/C/D/E): " 
        input keluhan1 keluhan2 keluhan3

        if ((keluhan1 <-- 'A' atau keluhan2 <--'A' atau keluhan3 <-- 'A') and
            (keluhan1 <-- 'B' atau keluhan2 <-- 'B' atau keluhan3 <--'B') and
            (keluhan1 <-- 'C' atau keluhan2 <-- 'C' atau keluhan3 <-- 'C')) then
        {
            output "Kamu memiliki penyakit Vertigo!" 
        } 
        else if ((keluhan1 <-- 'B' atau keluhan2 <-- 'B' atau keluhan3 <-- 'B') and
                 (keluhan1 <-- 'C' atau keluhan2 <-- 'C' atau keluhan3 <-- 'C') and
                 (keluhan1 <-- 'D' atau keluhan2 <-- 'D' atau keluhan3 <-- 'D')) then
        {
            output "Kamu memiliki penyakit Influenza Atau Flu!"
        } 
        else if ((keluhan1 <-- 'C' atau keluhan2 <-- 'C' atau keluhan3 <-- 'C') and
                 (keluhan1 <-- 'D' atau keluhan2 <-- 'D' atau keluhan3 <-- 'D') and
                 (keluhan1 <-- 'E' atau keluhan2 <-- 'E' atau keluhan3 <-- 'E')) then
        {
            output  "Kamu memiliki penyakit Demam Tifoid!"
        } 
        else if ((keluhan1 <-- 'A' atau keluhan2 <-- 'A' atau keluhan3 <-- 'A') and
                 (keluhan1 <-- 'D' atau keluhan2 <-- 'D' atau keluhan3 <-- 'D') and
                 (keluhan1 <-- 'E' atau keluhan2 <-- 'E  atau keluhan3 <-- 'E')) then
		{
            output  "Kamu memiliki penyakit Pneumonia!" 
        }
        else if ((keluhan1 <-- 'B' atau keluhan2 <-- 'B' atau keluhan3 == 'B') and
                 (keluhan1 <-- 'D' atau keluhan2 <-- 'D' atau keluhan3 == 'D') and
                 (keluhan1 <-- 'E' atau keluhan2 <-- 'E' atau keluhan3 == 'E')) then
		{
            output  "Kamu memiliki penyakit Kolitis atau Divertikulitis !" 
        }
        else
        {
            output  "Gejala tidak dikenali."
        }
        output "Apakah Anda ingin melakukan diagnosa lagi? (Y/N): " 
        input pilihan

} until (pilihan = 'Y' atau pilihan = 'y')

output "Terima kasih telah mengunjungi rumah sakit kami!
