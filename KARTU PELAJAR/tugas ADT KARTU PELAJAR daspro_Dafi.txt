//kamus ADT Kartu Pelajar
//TGL
type TGL
{
    tempat, BB <-- string
    HH, TT <-- int
}
//NAMA
type NAMA
{
    nama_pjg, gelarD, gelarB <-- string
}
//NOMOR NIS
type NONIS
{
    nomnis, nonis <-- long long int
}
//ALAMAT
type ALAMAT
{
    alamat, kelurahan, kecamatan <-- string
    rtrw <-- int
}
//MASA BERLAKU KARTU
type BERLAKU
{
    berlaku <-- string
}
//TGL KARTU DIBUAT

{
    tempatt, MM <-- string
    DD, YY <-- int
}
//NAMA SEKOLAH
type SEKOLAH
{
    sklh <-- string
}
//JENJANG PENDIDIKAN
type JENJANG
{
    jenpen <-- string
}
//YANG MENANDATANGANI
type PARAF
{
    namaparaf <-- string
}

}
