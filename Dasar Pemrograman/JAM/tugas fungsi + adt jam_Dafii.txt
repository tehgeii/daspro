ubah_format_jam_ke_detik(jam, menit, detik) <-- integer
{
	jam * 3600 + menit * 60 + detik <-- return
}
ngurangi_jam(waktudatang, waktuberangkat) <-- integer
{
	waktudatang - waktuberangkat <-- return
}
balik_ke_bentuk_jam(total_detik, jam, menit, detik)
{
	total_detik div 3600 <-- jam
	(total_detik mod 3600) div 60 <-- menit
	total_detik mod 60 <-- detik
}
outputan(total_detik, jam, menit, detik)
{
	output "Maka Total Waktu Perjalanan Anda Adalah: ", total_detik
	output "Atau ", jam, " Jam ", menit, " Menit ", detik, " Detik"
}
inputan_jam_berangkat(jb <-- jamberangkat)
{
	output "Masukkan Data Waktu Jam Datang Anda"
	output "Jam: "
	input jb.JB
	output "Menit: "
	input jb.MB
	output "Detik: "
	input jb.DB
}
inputan_jam_datang(jd <-- jamdatang)
{
	output "Masukkan Data Waktu Jam Datang Anda"

}
