//judul: if else simple
#include <iostream>
using namespace std;
//kamus
int angka1, angka2;
int main()
{
    cout << "contoh if else simple" << endl;
    cout << "masukkan angka 1: " << endl;
    cin >> angka1;
    cout << "masukkan angka 2: " << endl;
    cin >> angka2;
    if (angka1 > angka2)
    {
        cout << "angka 1 besar" << endl;
    }
    else
    {
        cout << "angka 2 besar" << endl;
    }
    return 0;
}
