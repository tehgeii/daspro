//kamus untuk KTP
KTP
{
    nama, nik, ttl, jk, golonganDarah, alamat, agama, status, pekerjaan, kewarganegaraan <-- string
}
//kamus untuk SIM
SIM
{
    nama, nomor, golonganDarah, alamat, waktuBerlaku, jenisKendaraan, pekerjaan <-- string
}

    input ktp.status
