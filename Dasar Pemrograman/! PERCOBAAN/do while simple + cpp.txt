//judul: do while simple
#include <iostream>
using namespace std;
//kamus
char pilihan;
int main()
{
    cout << "contoh do while (loop) simple" << endl;
    do
    {
        cout << "apakah anda ingin mengulang?" << endl;
        cout << "ketik (Y) untuk mengulang, ketik huruf apapun untuk tidak mengulang" << endl;
        cin >> pilihan;
    } while (pilihan == 'y' || pilihan == 'Y');
    cout << "anda keluar dari loop" << endl;
    return 0;
}
